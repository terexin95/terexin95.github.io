Подключаем виджет

			
				<script src="remarked.js"></script>
				<script>
					widgetRemarked({
						title: 'Название ресторана',
						color_red: '#a50204',
						email: true, 
						date: true, 
						qty: true, 
						time: true, 
						text: true 
					})
				</script>

* к нужной кнопке добавить класс remarked-open-widget

Настройки виджета

			
				<script>
					widgetRemarked({
						title: 'Название ресторана', // название в шапке
						color_red: '#a50204', //Красный цвет в виджете
						email: true, // Если написать false, то отключит поле email
						date: true, // Если написать false, то отключит поле с датой
						qty: true, // Если написать false, то отключит поле с количеством гостей
						time: true, // Если написать false, то отключит блок с выбором времени
						text: true // Если написать false, то отключит поле с коментарием
					})
				</script>
